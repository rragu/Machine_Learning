def recommend_user_items(productID, pivot_df, preds_df, num_recommendations):
      
    product_idx = productID-1 # index starts at 0
    
    # Get and sort the user's ratings
    sorted_product_ratings = pivot_df.iloc[product_idx].sort_values(ascending=False)
    #sorted_user_ratings
    sorted_product_predictions = preds_df.iloc[product_idx].sort_values(ascending=False)
    #sorted_user_predictions

    temp_pd = pd.concat([sorted_product_ratings, sorted_product_predictions], axis=1)
    temp_pd.index.name = 'Recommended Items'
    temp_pd.columns = ['item_ratings', 'item_predictions']
    
    temp_pd = temp_pd.loc[temp_pd.item_ratings == 0]   
    temp_pd = temp_pd.sort_values('item_predictions', ascending=False)
    print('\nBelow are the recommended items for product(product_id = {}):\n'.format(productID))
    print(temp_pd.head(num_recommendations))
